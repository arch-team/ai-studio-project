## 技术规范

### API 层模式

```typescript
// adminApi.ts
export async function fetchUsers(filters: UserFilters): Promise<UserListResponse> {
  const url = buildUrl('/users', filters);
  const response = await fetch(url, { headers: getAuthHeaders() });
  return handleResponse<UserListResponse>(response);
}

// queries.ts
export function useUsers(filters: UserFilters = {}) {
  return useQuery({
    queryKey: queryKeys.users.list(filters as Record<string, unknown>),
    queryFn: () => fetchUsers(filters),
  });
}
```
