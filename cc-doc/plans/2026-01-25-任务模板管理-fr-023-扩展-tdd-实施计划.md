```python
from enum import Enum

class TemplateVisibility(str, Enum):
    """模板可见性范围"""
    PRIVATE = "private"  # 仅自己可见
    TEAM = "team"        # 团队可见
    PUBLIC = "public"    # 所有人可见
```

#### 2. 领域实体 (`domain/entities/job_template.py`)

```python
@dataclass
class JobTemplate:
    """任务模板领域实体"""
    id: int
    name: str
    owner_id: int
    training_config: dict
    description: str | None = None
    visibility: TemplateVisibility = TemplateVisibility.PRIVATE
    usage_count: int = 0
    last_used_at: datetime | None = None
    created_at: datetime = field(default_factory=utc_now)
    updated_at: datetime = field(default_factory=utc_now)
    deleted_at: datetime | None = None

    def is_visible_to(self, user_id: int) -> bool:
        """检查用户是否可见此模板"""
        if self.visibility == TemplateVisibility.PUBLIC:
            return True
        return self.owner_id == user_id

    def can_modify(self, user_id: int) -> bool:
        """检查用户是否可修改"""
        return self.owner_id == user_id

    def increment_usage(self) -> None:
        """增加使用次数"""
        self.usage_count += 1
        self.last_used_at = utc_now()
        self.updated_at = utc_now()

    def is_deleted(self) -> bool:
        return self.deleted_at is not None
```

#### 3. 仓库接口 (`domain/repositories/job_template_repository.py`)

```python
class IJobTemplateRepository(ABC):
    @abstractmethod
    async def get_by_id(self, template_id: int) -> JobTemplate | None: ...

    @abstractmethod
    async def list_templates(
        self,
        owner_id: int | None = None,
        visibility: TemplateVisibility | None = None,
        search_name: str | None = None,
        include_deleted: bool = False,
        page: int = 1,
        page_size: int = 20,
        sort_by: str = "usage_count",
        sort_order: str = "desc",
    ) -> tuple[list[JobTemplate], int]: ...

    @abstractmethod
    async def list_visible_templates(
        self,
        user_id: int,
        page: int = 1,
        page_size: int = 20,
    ) -> tuple[list[JobTemplate], int]:
        """列出用户可见的所有模板"""

    @abstractmethod
    async def get_popular_templates(self, limit: int = 10) -> list[JobTemplate]: ...

    @abstractmethod
    async def create(self, template: JobTemplate) -> JobTemplate: ...

    @abstractmethod
    async def update(self, template: JobTemplate) -> JobTemplate: ...

    @abstractmethod
    async def soft_delete(self, template_id: int) -> bool: ...

    @abstractmethod
    async def increment_usage_count(self, template_id: int) -> None: ...
```

#### 5. 仓库实现 (`infrastructure/repositories/job_template_repository_impl.py`)

```python
class JobTemplateRepository(EnhancedBaseRepository[JobTemplateModel, JobTemplate]):
    """JobTemplate 仓库实现"""

    def __init__(self, session: AsyncSession):
        super().__init__(session, JobTemplateModel, JobTemplate)

    async def list_visible_templates(
        self,
        user_id: int,
        page: int = 1,
        page_size: int = 20,
    ) -> tuple[list[JobTemplate], int]:
        """列出用户可见的模板 (自己的 + 公开的)"""
        query = select(JobTemplateModel).where(
            and_(
                JobTemplateModel.deleted_at.is_(None),
                or_(
                    JobTemplateModel.owner_id == user_id,
                    JobTemplateModel.visibility == TemplateVisibility.PUBLIC,
                )
            )
        ).order_by(JobTemplateModel.usage_count.desc())
        # 分页逻辑...

    async def get_popular_templates(self, limit: int = 10) -> list[JobTemplate]:
        """获取热门模板"""
        query = select(JobTemplateModel).where(
            and_(
                JobTemplateModel.deleted_at.is_(None),
                JobTemplateModel.visibility == TemplateVisibility.PUBLIC,
            )
        ).order_by(JobTemplateModel.usage_count.desc()).limit(limit)
        # ...

    async def increment_usage_count(self, template_id: int) -> None:
        """原子增加使用次数"""
        stmt = (
            update(JobTemplateModel)
            .where(JobTemplateModel.id == template_id)
            .values(
                usage_count=JobTemplateModel.usage_count + 1,
                last_used_at=utc_now(),
            )
        )
        await self._session.execute(stmt)
```

#### 3. Application Service (`application/services/job_template_service.py`)

```python
class JobTemplateService:
    def __init__(
        self,
        template_repository: IJobTemplateRepository,
        job_repository: ITrainingJobRepository,
    ):
        self._template_repo = template_repository
        self._job_repo = job_repository

    async def create_template(self, owner_id: int, data: dict) -> JobTemplate:
        template = JobTemplate(
            id=0,  # 由数据库生成
            name=data["name"],
            owner_id=owner_id,
            training_config=data["training_config"],
            description=data.get("description"),
            visibility=data.get("visibility", TemplateVisibility.PRIVATE),
        )
        return await self._template_repo.create(template)

    async def get_template(self, template_id: int, user_id: int) -> JobTemplate:
        template = await self._template_repo.get_by_id(template_id)
        if not template:
            raise EntityNotFoundError("JobTemplate", str(template_id))
        if not template.is_visible_to(user_id):
            raise PermissionDeniedError("无权访问此模板")
        return template

    async def update_template(self, template_id: int, user_id: int, data: dict) -> JobTemplate:
        template = await self._template_repo.get_by_id(template_id)
        if not template:
            raise EntityNotFoundError("JobTemplate", str(template_id))
        if not template.can_modify(user_id):
            raise PermissionDeniedError("无权修改此模板")
        # 更新字段...
        return await self._template_repo.update(template)

    async def delete_template(self, template_id: int, user_id: int) -> None:
        template = await self._template_repo.get_by_id(template_id)
        if not template:
            raise EntityNotFoundError("JobTemplate", str(template_id))
        if not template.can_modify(user_id):
            raise PermissionDeniedError("无权删除此模板")
        await self._template_repo.soft_delete(template_id)

    async def create_job_from_template(
        self,
        template_id: int,
        user_id: int,
        overrides: dict,
    ) -> TrainingJob:
        template = await self.get_template(template_id, user_id)

        # 合并模板配置和覆盖配置
        config = template.training_config.copy()
        # ... 合并 overrides

        # 创建训练任务
        job = TrainingJob(
            id=0,
            job_name=overrides["job_name"],
            owner_id=user_id,
            image_uri=config["image"],
            instance_type=config["instance_type"],
            node_count=overrides.get("node_count", config.get("instance_count", 1)),
            # ...
        )

        # 增加模板使用次数
        await self._template_repo.increment_usage_count(template_id)

        return await self._job_repo.create(job)
```

#### 4. API 端点 (`api/job_templates.py`)

```python
router = APIRouter(prefix="/job-templates", tags=["Job Templates"])

@router.post("", response_model=JobTemplateDetail, status_code=201)
async def create_template(
    request: CreateJobTemplateRequest,
    current_user: CurrentUser = Depends(require_engineer),
    service: JobTemplateService = Depends(get_job_template_service),
):
    template = await service.create_template(current_user.id, request.model_dump())
    return JobTemplateDetail.from_entity(template)

@router.get("", response_model=JobTemplateListResponse)
async def list_templates(
    visibility: TemplateVisibility | None = None,
    search: str | None = None,
    page: int = Query(1, ge=1),
    page_size: int = Query(20, ge=1, le=100),
    current_user: CurrentUser = Depends(require_viewer),
    service: JobTemplateService = Depends(get_job_template_service),
):
    items, total = await service.list_visible_templates(
        user_id=current_user.id,
        visibility=visibility,
        search_name=search,
        page=page,
        page_size=page_size,
    )
    return JobTemplateListResponse(
        items=[JobTemplateSummary.from_entity(t) for t in items],
        total=total,
        page=page,
        page_size=page_size,
    )

@router.get("/popular", response_model=list[JobTemplateSummary])
async def get_popular_templates(
    limit: int = Query(10, ge=1, le=50),
    service: JobTemplateService = Depends(get_job_template_service),
):
    templates = await service.get_popular_templates(limit)
    return [JobTemplateSummary.from_entity(t) for t in templates]

@router.get("/{template_id}", response_model=JobTemplateDetail)
async def get_template(
    template_id: int,
    current_user: CurrentUser = Depends(require_viewer),
    service: JobTemplateService = Depends(get_job_template_service),
):
    template = await service.get_template(template_id, current_user.id)
    return JobTemplateDetail.from_entity(template)

@router.delete("/{template_id}", status_code=204)
```

#### 3. TanStack Query Hooks (`api/queries.ts`)

```typescript
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { templateApi } from './templateApi';

export const templateKeys = {
  all: ['templates'] as const,
  lists: () => [...templateKeys.all, 'list'] as const,
  list: (filters: object) => [...templateKeys.lists(), filters] as const,
  popular: () => [...templateKeys.all, 'popular'] as const,
  details: () => [...templateKeys.all, 'detail'] as const,
  detail: (id: number) => [...templateKeys.details(), id] as const,
};

export function useTemplates(params?: { visibility?: string; search?: string; page?: number }) {
  return useQuery({
    queryKey: templateKeys.list(params ?? {}),
    queryFn: () => templateApi.list(params),
  });
}

export function usePopularTemplates(limit = 10) {
  return useQuery({
    queryKey: templateKeys.popular(),
    queryFn: () => templateApi.getPopular(limit),
  });
}

export function useTemplate(id: number) {
  return useQuery({
    queryKey: templateKeys.detail(id),
    queryFn: () => templateApi.getById(id),
    enabled: id > 0,
  });
}

export function useCreateTemplate() {
  const queryClient = useQueryClient();
  return useMutation({
    mutationFn: templateApi.create,
    onSuccess: () => queryClient.invalidateQueries({ queryKey: templateKeys.lists() }),
  });
}

export function useCreateJobFromTemplate() {
  const queryClient = useQueryClient();
  return useMutation({
    mutationFn: ({ templateId, data }: { templateId: number; data: CreateJobFromTemplateRequest }) =>
      templateApi.createJobFromTemplate(templateId, data),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: templateKeys.lists() });
      queryClient.invalidateQueries({ queryKey: ['trainingJobs'] });
    },
  });
}
```

#### 4. 模板列表页面 (`pages/TemplateListPage.tsx`)

```tsx
import { useState } from 'react';
import {
  Table, Header, Pagination, TextFilter, Select, SpaceBetween, Button, Badge
} from '@cloudscape-design/components';
import { useTemplates } from '../api/queries';
import type { JobTemplate, TemplateVisibility } from '../types';

export function TemplateListPage() {
  const [page, setPage] = useState(1);
  const [visibility, setVisibility] = useState<TemplateVisibility | undefined>();
  const [search, setSearch] = useState('');

  const { data, isLoading } = useTemplates({ visibility, search, page });

  const columns = [
    { id: 'name', header: '名称', cell: (item: JobTemplate) => item.name },
    { id: 'visibility', header: '可见性', cell: (item: JobTemplate) => (
      <Badge color={item.visibility === 'public' ? 'green' : 'grey'}>
        {item.visibility}
      </Badge>
    )},
    { id: 'usage_count', header: '使用次数', cell: (item: JobTemplate) => item.usage_count },
    { id: 'instance_type', header: '实例类型', cell: (item: JobTemplate) => item.training_config.instance_type },
    { id: 'actions', header: '操作', cell: (item: JobTemplate) => (
      <Button variant="link" href={`/training-jobs/create?template=${item.id}`}>使用</Button>
    )},
  ];

  return (
    <Table
      header={
        <Header
          actions={<Button variant="primary" href="/job-templates/create">创建模板</Button>}
        >
          任务模板
        </Header>
      }
      items={data?.items ?? []}
      columns={columns}
      loading={isLoading}
      filter={
        <SpaceBetween direction="horizontal" size="s">
          <TextFilter filteringText={search} onChange={({ detail }) => setSearch(detail.filteringText)} />
          <Select
            selectedOption={visibility ? { value: visibility } : null}
            onChange={({ detail }) => setVisibility(detail.selectedOption?.value as TemplateVisibility)}
            options={[
              { value: '', label: '全部' },
              { value: 'private', label: '私有' },
              { value: 'public', label: '公开' },
            ]}
          />
        </SpaceBetween>
      }
      pagination={
        <Pagination
          currentPageIndex={page}
          pagesCount={Math.ceil((data?.total ?? 0) / 20)}
          onChange={({ detail }) => setPage(detail.currentPageIndex)}
        />
      }
    />
  );
}
```
