class CheckpointMigrationService:
    """检查点分层迁移服务"""

    def __init__(
        self,
        checkpoint_repository: ICheckpointRepository,
        storage_service: IStorageService,
        notification_service: INotificationService,
    ):
        self._checkpoint_repo = checkpoint_repository
        self._storage = storage_service
        self._notification = notification_service

    async def run_migration_cycle(self) -> MigrationResult:
        """执行迁移周期 (定时任务每 10 分钟调用)"""

    async def migrate_checkpoint(
        self, checkpoint_id: int, target_tier: StorageTier
    ) -> Checkpoint:
        """迁移单个检查点到目标存储层"""

    async def handle_storage_pressure(
        self, tier: StorageTier, usage_percent: float
    ) -> None:
        """处理存储压力 (>90% 触发紧急迁移)"""

    async def verify_checkpoint_integrity(
        self, checkpoint_id: int
    ) -> bool:
        """验证检查点完整性"""
```

# storage_stack.py
from aws_cdk import Duration
from aws_cdk.aws_s3 import LifecycleRule, StorageClass, TagFilter

class StorageStack(Stack):
    def __init__(self, ...):
        # 现有配置...

        # 检查点生命周期规则
        self.checkpoint_bucket.add_lifecycle_rule(
            id="ColdCheckpointTransitionToIA",
            transitions=[
                Transition(
                    storage_class=StorageClass.INFREQUENT_ACCESS,
                    transition_after=Duration.days(
                        self.node.try_get_context("checkpoint_ia_transition_days") or 30
                    ),
                )
            ],
            tag_filters=[TagFilter(key="checkpoint_type", value="cold")],
        )

        self.checkpoint_bucket.add_lifecycle_rule(
            id="ColdCheckpointDeletion",
            expiration=Duration.days(
                self.node.try_get_context("checkpoint_retention_days") or 90
            ),
            tag_filters=[TagFilter(key="checkpoint_type", value="cold")],
        )
```

## 风险和注意事项

1. **HyperPod 环境依赖**: T038c 集成测试需要真实的 HyperPod 集群
2. **存储迁移性能**: 大文件迁移可能影响训练性能，需在空闲时段执行
3. **Model Registry 权限**: 需要 IAM 权限配置 SageMaker Model Registry 访问
4. **抢占超时处理**: 5 分钟内未完成检查点保存将强制抢占
