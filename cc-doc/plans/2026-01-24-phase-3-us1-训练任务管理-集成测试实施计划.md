```python
# backend/tests/integration/training/test_preemption_timing.py

"""抢占时序 SLA 集成测试 (T038c)

测试场景:
1. 低优先级任务被高优先级任务抢占 (Kueue Priority)
2. checkpoint 在抢占触发后 5 分钟内保存完成
3. 被抢占 Pod 在 30 秒内释放
4. 任务状态正确转换为 Preempted
5. 抢占后自动恢复成功

依赖: T022 (checkpoints 表), T024 (Checkpoint 模型), T029 (状态同步), T038 (checkpoint 保存)
"""

import pytest
from datetime import datetime
from decimal import Decimal
from unittest.mock import AsyncMock, MagicMock, patch
import time

from src.modules.training.application.services.checkpoint_service import (
    CheckpointService,
)
from src.modules.training.application.services.training_sync_service import (
    TrainingSyncService,
)
from src.modules.training.domain.entities import TrainingJob, Checkpoint
from src.modules.training.domain.value_objects import (
    JobStatus,
    JobPriority,

)


def create_test_job(
    job_id: int = 1,
    priority: JobPriority = JobPriority.MEDIUM,
    status: JobStatus = JobStatus.RUNNING,
) -> TrainingJob:
    """创建测试用 TrainingJob"""
    return TrainingJob(
        id=job_id,
        job_name=f"test-job-{job_id}",
        owner_id=100,
        image_uri="123456.dkr.ecr.us-west-2.amazonaws.com/pytorch:2.1",
        instance_type="ml.p4d.24xlarge",
        entrypoint_command=["torchrun", "--nproc_per_node=8", "train.py"],
        node_count=2,
        tasks_per_node=8,
        distribution_strategy=DistributionStrategy.DDP,
        priority=priority,
        status=status,
        preemption_count=0,
        created_at=datetime.utcnow(),
        updated_at=datetime.utcnow(),
    )

def mock_checkpoint_repository():
    """Mock ICheckpointRepository"""

    service.get_checkpoint_size = AsyncMock(return_value=1024 * 1024 * 100)  # 100MB
    return service


@pytest.mark.integration
class TestPreemptionTimingSLA:
    """抢占时序 SLA 集成测试 - FR-004"""

    @pytest.mark.asyncio
    async def test_low_priority_job_preempted_by_high_priority(
        self,
        mock_training_job_repository,
        mock_hyperpod_client,
    ):
        """场景1: 低优先级任务被高优先级任务抢占"""
        # Arrange: 创建低优先级 Running 任务
        low_priority_job = create_test_job(
            job_id=1, priority=JobPriority.LOW, status=JobStatus.RUNNING
        )
        mock_hyperpod_client.get_training_job_status.return_value = {
            "job_name": low_priority_job.job_name,
            "status": "Preempted",
        }

        service = TrainingSyncService(
            training_job_repository=mock_training_job_repository,
            hyperpod_client=mock_hyperpod_client,
            cluster_name="test-cluster",
        )

        # Act: 模拟高优先级任务触发抢占
        await service.sync_job(low_priority_job)

        # Assert: 验证状态转换为 PREEMPTED
        assert low_priority_job.status == JobStatus.PREEMPTED
        assert low_priority_job.preemption_count == 1

    @pytest.mark.asyncio
    async def test_checkpoint_saved_within_5_minutes(
        self,
        mock_checkpoint_repository,
        mock_training_job_repository,
        mock_storage_service,
    ):
        """场景2: checkpoint 在抢占触发后 5 分钟内保存完成"""
        # Arrange
        job = create_test_job(status=JobStatus.RUNNING)
        mock_training_job_repository.get_by_id.return_value = job

        checkpoint_service = CheckpointService(
            repository=mock_checkpoint_repository,
            training_job_repository=mock_training_job_repository,
            storage_service=mock_storage_service,
        )

        # Act: 记录开始时间，创建抢占检查点
        start_time = time.time()
        checkpoint = await checkpoint_service.create_checkpoint_on_preemption(
            job_id=job.id,
            timeout_seconds=300,  # 5 分钟 SLA
        )
        elapsed = time.time() - start_time

        # Assert: 验证在 5 分钟内完成
        assert checkpoint is not None
        assert elapsed < 300  # 5 分钟
        mock_checkpoint_repository.create.assert_called_once()

    @pytest.mark.asyncio
    async def test_pod_released_within_30_seconds(
        self,
        mock_hyperpod_client,
    ):
        """场景3: 被抢占 Pod 在 30 秒内释放"""
        # Arrange: 模拟 Kueue Evicted 后 Pod 状态
        mock_hyperpod_client.get_pod_status.return_value = {
            "phase": "Terminated",
            "termination_time": datetime.utcnow(),
        }

        # Act: 查询 Pod 状态
        start_time = time.time()
        pod_status = await mock_hyperpod_client.get_pod_status("test-pod")
        elapsed = time.time() - start_time

        # Assert: 验证 Pod 已终止，响应时间 < 30s
        assert pod_status["phase"] == "Terminated"
        assert elapsed < 30

    @pytest.mark.asyncio
    async def test_job_status_transitions_to_preempted(
        self,
        mock_training_job_repository,
        mock_hyperpod_client,
    ):
        """场景4: 任务状态正确转换为 Preempted"""
        # Arrange
        job = create_test_job(status=JobStatus.RUNNING)
        mock_hyperpod_client.get_training_job_status.return_value = {
            "job_name": job.job_name,
            "status": "Preempted",
        }

        service = TrainingSyncService(
            training_job_repository=mock_training_job_repository,
            hyperpod_client=mock_hyperpod_client,
            cluster_name="test-cluster",
        )

        # Act
        await service.sync_job(job)

        # Assert: 验证状态机正确转换
        assert job.status == JobStatus.PREEMPTED
        assert job.can_transition_to(JobStatus.RUNNING)  # 可以恢复
        assert not job.is_terminal()  # 非终态

    @pytest.mark.asyncio
    async def test_auto_recovery_from_checkpoint(
        self,
        mock_checkpoint_repository,
        mock_training_job_repository,
        mock_hyperpod_client,
    ):
        """场景5: 抢占后自动恢复成功 (checkpoint 恢复 + 训练继续)"""
        # Arrange: 任务处于 PREEMPTED 状态，有可用检查点
        job = create_test_job(status=JobStatus.PREEMPTED)
        job.preemption_count = 1

        mock_checkpoint = Checkpoint(
            id=1,
            training_job_id=job.id,
            checkpoint_name="preemption-20260124120000",
            storage_path="/mnt/nvme/ckpt/1/preemption",
            size_bytes=100 * 1024 * 1024,
            trigger_type=CheckpointTriggerType.PREEMPTION,
        )
        mock_checkpoint_repository.get_latest_by_training_job_id.return_value = (
            mock_checkpoint
        )

        # 模拟恢复后任务变为 Running
        mock_hyperpod_client.get_training_job_status.return_value = {
            "job_name": job.job_name,
            "status": "Running",
        }

        service = TrainingSyncService(
            training_job_repository=mock_training_job_repository,
            hyperpod_client=mock_hyperpod_client,
            cluster_name="test-cluster",
        )

        # Act: 模拟重新调度后状态同步
        job.status = JobStatus.RUNNING  # 模拟 Kueue 重新调度
        await service.sync_job(job)

        # Assert: 验证恢复成功
        assert job.status == JobStatus.RUNNING
        # preemption_count 保持不变 (成功恢复后不重置)
```
