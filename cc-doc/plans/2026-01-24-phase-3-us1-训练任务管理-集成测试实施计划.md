```python
@pytest.mark.integration
class TestPreemptionTimingSLA:
    """抢占时序 SLA 集成测试 - FR-004"""

    @pytest.mark.asyncio
    async def test_low_priority_job_preempted_by_high_priority(
        self,
        mock_training_job_repository,
        mock_hyperpod_client,
    ):
        """场景1: 低优先级任务被高优先级任务抢占"""
        # Arrange: 创建低优先级 Running 任务
        low_priority_job = create_test_job(
            job_id=1, priority=JobPriority.LOW, status=JobStatus.RUNNING
        )
        mock_hyperpod_client.get_training_job_status.return_value = {
            "job_name": low_priority_job.job_name,
            "status": "Preempted",
        }

        service = TrainingSyncService(
            training_job_repository=mock_training_job_repository,
            hyperpod_client=mock_hyperpod_client,
            cluster_name="test-cluster",
        )

        # Act: 模拟高优先级任务触发抢占
        await service.sync_job(low_priority_job)

        # Assert: 验证状态转换为 PREEMPTED
        assert low_priority_job.status == JobStatus.PREEMPTED
        assert low_priority_job.preemption_count == 1

    @pytest.mark.asyncio
    async def test_checkpoint_saved_within_5_minutes(
        self,
        mock_checkpoint_repository,
        mock_training_job_repository,
        mock_storage_service,
    ):
        """场景2: checkpoint 在抢占触发后 5 分钟内保存完成"""
        # Arrange
        job = create_test_job(status=JobStatus.RUNNING)
        mock_training_job_repository.get_by_id.return_value = job

        checkpoint_service = CheckpointService(
            repository=mock_checkpoint_repository,
            training_job_repository=mock_training_job_repository,
            storage_service=mock_storage_service,
        )

        # Act: 记录开始时间，创建抢占检查点
        start_time = time.time()
        checkpoint = await checkpoint_service.create_checkpoint_on_preemption(
            job_id=job.id,
            timeout_seconds=300,  # 5 分钟 SLA
        )
        elapsed = time.time() - start_time

        # Assert: 验证在 5 分钟内完成
        assert checkpoint is not None
        assert elapsed < 300  # 5 分钟
        mock_checkpoint_repository.create.assert_called_once()

    @pytest.mark.asyncio
    async def test_pod_released_within_30_seconds(
        self,
        mock_hyperpod_client,
    ):
        """场景3: 被抢占 Pod 在 30 秒内释放"""
        # Arrange: 模拟 Kueue Evicted 后 Pod 状态
        mock_hyperpod_client.get_pod_status.return_value = {
            "phase": "Terminated",
            "termination_time": datetime.utcnow(),

        # Act: 查询 Pod 状态
        start_time = time.time()
        pod_status = await mock_hyperpod_client.get_pod_status("test-pod")
        elapsed = time.time() - start_time

        # Assert: 验证 Pod 已终止，响应时间 < 30s
        assert pod_status["phase"] == "Terminated"
        assert elapsed < 30

    @pytest.mark.asyncio
    async def test_job_status_transitions_to_preempted(
        self,
        mock_training_job_repository,
        mock_hyperpod_client,
    ):
        """场景4: 任务状态正确转换为 Preempted"""
        # Arrange
        job = create_test_job(status=JobStatus.RUNNING)
        mock_hyperpod_client.get_training_job_status.return_value = {
            "job_name": job.job_name,
            "status": "Preempted",
        }

            training_job_repository=mock_training_job_repository,
            hyperpod_client=mock_hyperpod_client,
            cluster_name="test-cluster",

        assert job.status == JobStatus.PREEMPTED
        assert job.can_transition_to(JobStatus.RUNNING)  # 可以恢复
        assert not job.is_terminal()  # 非终态

    @pytest.mark.asyncio
    async def test_auto_recovery_from_checkpoint(
        self,
        mock_checkpoint_repository,
        mock_training_job_repository,
        mock_hyperpod_client,

            id=1,
            training_job_id=job.id,
            checkpoint_name="preemption-20260124120000",

            mock_checkpoint
        )

        mock_hyperpod_client.get_training_job_status.return_value = {
            "job_name": job.job_name,
            "status": "Running",

            training_job_repository=mock_training_job_repository,
            hyperpod_client=mock_hyperpod_client,
            cluster_name="test-cluster",
        )

        job.status = JobStatus.RUNNING  # 模拟 Kueue 重新调度

        # Assert: 验证恢复成功
        assert job.status == JobStatus.RUNNING
        # preemption_count 保持不变 (成功恢复后不重置)
```

```python
    async def _wait_for_status(
        self,
        client,
        job_id: str,
        expected_status: str,
        timeout: int,
    ) -> str:
        """等待任务达到指定状态"""
        start = time.time()
        while time.time() - start < timeout:
            status = await client.get_training_job_status(job_id)
            if status["status"] == expected_status:
                return expected_status
            await asyncio.sleep(5)
        raise TimeoutError(
            f"Job {job_id} did not reach {expected_status} within {timeout}s"
        )

    async def _wait_for_api_status(
        self,
        client,
        token: str,
        job_id: int,
        expected_status: str,
        timeout: int,
    ) -> str:
        """通过 API 等待任务状态"""
        start = time.time()
        while time.time() - start < timeout:
            response = await client.get(
                f"/api/v1/training-jobs/{job_id}",
                headers={"Authorization": f"Bearer {token}"},
            )
            if response.json()["status"] == expected_status:
                return expected_status
            await asyncio.sleep(5)
        raise TimeoutError(
            f"Job {job_id} did not reach {expected_status} within {timeout}s"
        )

    async def _wait_for_checkpoint(
        self,
        client,
        job_id: str,
        timeout: int,
    ) -> dict:
        """等待 checkpoint 完成"""
        start = time.time()
        while time.time() - start < timeout:
            checkpoints = await client.list_checkpoints(job_id)
            preemption_ckpts = [
                c for c in checkpoints
                if c["trigger_type"] == "PREEMPTION"
            ]
            if preemption_ckpts:
                return preemption_ckpts[-1]
            await asyncio.sleep(10)
        return None

    async def _wait_for_pod_terminated(
        self,
        client,
        pod_name: str,
        timeout: int,
    ):
        """等待 Pod 终止"""
        start = time.time()
        while time.time() - start < timeout:
            pod_status = await client.get_pod_status(pod_name)
            if pod_status.get("phase") in ["Terminated", "Failed", "Succeeded"]:
                return
            await asyncio.sleep(2)
        raise TimeoutError(f"Pod {pod_name} not terminated within {timeout}s")

    async def _cleanup_jobs(self, client, job_ids: list):
        """清理测试任务"""
        for job_id in job_ids:
            if job_id:
                try:
                    await client.cancel_training_job(job_id)
                except Exception as e:
                    print(f"⚠️ 清理任务 {job_id} 失败: {e}")
```

### 7.6 资源清理策略

```python
# backend/tests/e2e/aws/conftest.py (续)

import atexit
from typing import List

# ============================================================
# 全局资源跟踪和清理
# ============================================================

_created_resources: List[dict] = []

def track_resource(resource_type: str, resource_id: str):
    """跟踪创建的资源"""
    _created_resources.append({
        "type": resource_type,
        "id": resource_id,
        "created_at": datetime.utcnow(),
    })

async def cleanup_all_resources(hyperpod_client):
    """清理所有创建的资源"""
    for resource in _created_resources:
        try:
            if resource["type"] == "training_job":
                await hyperpod_client.cancel_training_job(resource["id"])
            elif resource["type"] == "checkpoint":
                await hyperpod_client.delete_checkpoint(resource["id"])
            print(f"✅ 清理 {resource['type']}: {resource['id']}")
        except Exception as e:
            print(f"⚠️ 清理失败 {resource['type']}: {resource['id']}: {e}")
    _created_resources.clear()

@pytest.fixture(scope="session", autouse=True)
async def cleanup_on_exit(hyperpod_client):
    """Session 结束时清理所有资源"""
    yield
    await cleanup_all_resources(hyperpod_client)
```
